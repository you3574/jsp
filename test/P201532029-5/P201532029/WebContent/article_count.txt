CREATE DEFINER=`user1`@`localhost` PROCEDURE `article_count`(
  IN ss VARCHAR(10),
  IN st NVARCHAR(100)
)
BEGIN


SELECT COUNT(a.ID)
FROM article a
  LEFT JOIN user u ON a.userId = u.id  
WHERE 
  (ss = '0') OR
  (ss = '1' AND name LIKE concat(st,'%')) OR
  (ss = '2' AND title LIKE concat('%',st,'%'));

END
