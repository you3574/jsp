CREATE DEFINER=`root`@`localhost` PROCEDURE `article_findAll`(
  IN firstRecordIndex INT,
  IN pageSize INT,
  IN ss VARCHAR(10),
  IN st NVARCHAR(100),
  IN od VARCHAR(10)
)
BEGIN

SELECT a.id, a.no, b.boardName, u.name,a. writeTime, a.title
FROM user u, article a, board b
WHERE 
  (ss = '0') OR
  (ss = '1','2' AND name LIKE st) OR
  (ss = '3' AND title LIKE st) OR
  (ss = '4' AND no like st)
ORDER BY 
  CASE WHEN od = '0' THEN a.id
  END,
  CASE WHEN od = '1' THEN u.name
  END,
  CASE WHEN od = '4' THEN a.title
  END DESC,
  CASE WHEN od ='3' THEN a.no
  END,
  CASE WHEN od= '2' THEN u.name
  END DESC
LIMIT firstRecordIndex, pageSize;
  
END
